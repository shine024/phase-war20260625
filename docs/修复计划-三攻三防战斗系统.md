## 一、问题概述

### 1.1 设定文档要求

根据最新设定的战斗系统，战斗需要实现：

| 系统组件 | 要求 |
|---------|------|
| **索敌系统** | 炮类（SIEGE）和飞机（AIR）从远到近，其他单位从近到远 |
| **三攻系统** | 对轻攻击、对甲攻击、对空攻击 |
| **攻击方式** | 直射、曲射、空射 |
| **三防系统** | 对直射防御、对曲射防御、对空射防御 |
| **伤害计算** | 攻击力 × (1 - 对应防御减伤率) |

### 1.2 当前系统差距

| 差距项 | 现状 | 目标 |
|--------|------|------|
| 索敌方式 | 全部从近到远 | 根据平台类型区分索敌方向 |
| 三攻定义 | 已实现 | 保持现状 |
| 三防定义 | 名称为 defense_light/armor/air（防目标类型） | 改为对武器类型的防御 |
| 攻防匹配 | 攻击目标类型 ↔ 防御目标类型 | 攻击武器类型 ↔ 防御武器类型 |
| 平台索敌映射 | 无 | 需要建立映射表 |

### 3.2 阶段二：索敌系统重构

#### 3.2.1 修改 `scripts/combat_targeting.gd`

**目标：** 实现从近到远和从远到近两种索敌方式

```gdscript
## 根据索敌方式查找目标
## @param origin: 攻击者位置
## @param attacker_is_player: 是否玩家单位
## @param battle_manager: 战斗管理器
## @param max_range: 最大射程
## @param targeting_mode: 索敌方式 (NEAREST_FIRST / FARTHEST_FIRST)
## @param attacker_stats: 攻击者属性（用于获取平台类型）
static func find_targets_with_mode(
    origin: Vector2,
    attacker_is_player: bool,
    battle_manager: Node,
    max_range: float = 2400.0,
    targeting_mode: int = GC.TargetingMode.NEAREST_FIRST,
    attacker_stats: UnitStats = null
) -> Array:
    var grp: String = opponent_unit_group(attacker_is_player)
    var nodes: Array = battle_manager.get_cached_nodes_in_group(grp)

    var candidates: Array = []
    var cap_sq: float = max_range * max_range if max_range > 0.0 else 1e18

    for node in nodes:
        if not is_attackable_combat_unit(node):
            continue
        var n2d: Node2D = node as Node2D
        var d2: float = origin.distance_squared_to(n2d.global_position)
        if d2 <= cap_sq:
            candidates.append({"node": n2d, "distance_sq": d2})

    # 根据索敌方式排序
    if targeting_mode == GC.TargetingMode.FARTHEST_FIRST:
        # 从远到近：距离最大的优先
        candidates.sort_custom(func(a, b): return a.distance_sq > b.distance_sq)
    else:
        # 从近到远：距离最小的优先（默认）
        candidates.sort_custom(func(a, b): return a.distance_sq < b.distance_sq)

    return candidates
```

**工作量：** 2 工时
**优先级：** P0

#### 3.2.2 修改 `scripts/battle/construct_unit_ai.gd`

**目标：** 在 `find_target` 函数中应用索敌方式

```gdscript
static func find_target(u: CharacterBody2D, _delta: float) -> void:
    if should_retain_current_target(u):
        return
    u.target = null

    # 获取索敌方式
    var targeting_mode: int = GC.TargetingMode.NEAREST_FIRST
    if u.stats != null:
        # 通过 combat_kind 推断平台类型
        var platform_type: int = _infer_platform_type_from_combat_kind(u.stats.combat_kind)
        targeting_mode = GC.get_targeting_mode_for_platform(platform_type)

    # 卡牌网格战斗
    if GameManager and GameManager.is_card_grid_battle():
        var slot_target = _find_target_by_card_grid_with_mode(u, targeting_mode)
        if slot_target != null:
            u.target = slot_target
            return

    # 传统战场：使用空间分区系统
    if BattleManager and BattleManager.spatial_grid:
        var spatial_grid = BattleManager.spatial_grid
        if spatial_grid:
            var max_range: float = mini(acquisition_range(u), 250.0)
            # TODO: spatial_grid 需要支持索敌方式参数
            var nearest_target = spatial_grid.query_nearest_target(
                u.global_position,
                u.is_player,
                max_range,
                targeting_mode
            )
            if nearest_target != null:
                u.target = nearest_target
                return

    # 回退到传统方法（使用新索敌函数）
    var tree = u.get_tree()
    if tree == null:
        return

    var target_group: String = "enemy_units" if u.is_player else "player_units"
    var gr: Array = BattleManager.get_cached_nodes_in_group(target_group)

    var candidates = CombatTargeting.find_targets_with_mode(
        u.global_position,
        u.is_player,
        BattleManager,
        acquisition_range(u),
        targeting_mode,
        u.stats
    )

    # 选择最终目标（优先应用 TargetSelection 逻辑）
    if not candidates.is_empty():
        var candidate_nodes: Array = []
        for c in candidates:
            candidate_nodes.append(c.node)

        if u.stats != null:
            var selected: Node2D = TargetSelection.select_target(
                u, candidate_nodes, u.stats.weapon_type
            )
            if selected != null:
                u.target = selected
                return

        u.target = candidates[0].node as Node2D
        return

    # 无对方战斗单位时，攻击对方相位场
    if targeting_opponent_phase_field_only(u):
        var phase_field: Node2D = CombatTargeting.find_opponent_phase_field(
            u.global_position, u.is_player, BattleManager, -1.0
        )
        if phase_field != null:
            u.target = phase_field
            return

## 辅助函数：从 combat_kind 推断平台类型
static func _infer_platform_type_from_combat_kind(combat_kind: int) -> int:
    match combat_kind:
        GC.CombatKind.AIR:
            return GC.PlatformType.AIR
        GC.CombatKind.SUPPORT:
            # 支援单位中，火炮类归为 SIEGE
            # 这里需要更多上下文，暂时使用 FORTRESS
            return GC.PlatformType.SIEGE
        _:
            return GC.PlatformType.GUARD  # 默认
```

@export var defense_aerial: float = 0.0

# 存档迁移：从旧属性读取到新属性
func migrate_defense_from_legacy() -> void:
    # 如果旧属性存在（通过反射检测），迁移到新属性
    if has_meta("legacy_defense_light"):
        defense_direct = get_meta("legacy_defense_light")
    if has_meta("legacy_defense_armor"):
        defense_indirect = get_meta("legacy_defense_armor")
    if has_meta("legacy_defense_air"):
        defense_aerial = get_meta("legacy_defense_air")

### 6.1 技术风险

| 风险项 | 风险等级 | 影响 | 缓解措施 |
|--------|---------|------|---------|
| 存档不兼容 | 中 | 旧存档无法读取 | 实现向后兼容层 |
| 性能下降 | 低 | 索敌排序可能增加开销 | 限制候选数量，使用空间分区 |
| 逻辑错误 | 高 | 攻防匹配错误导致伤害异常 | 充分的单元测试，灰度发布 |
| 数据遗漏 | 中 | 某些卡牌缺少三防属性 | 批量检查脚本，默认值兜底 |

### 6.2 业务风险

| 风险项 | 风险等级 | 影响 | 缓解措施 |
|--------|---------|------|---------|
| 平衡性变化 | 高 | 索敌方式改变导致策略变化 | 逐步调整数值，观察玩家反馈 |
| 玩家困惑 | 中 | 玩家不理解新的攻防机制 | 添加教学关卡，更新UI说明 |
| 体验下降 | 低 | 新机制增加复杂度 | 保持UI简洁，自动索敌无需玩家操作 |
